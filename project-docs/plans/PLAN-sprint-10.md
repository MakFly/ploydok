## PR 2 — Keyring + API crypto

### `apps/api/src/signing/keyring.ts`

```typescript
// SPDX-License-Identifier: AGPL-3.0-only

import { readFile, writeFile, chmod } from "fs/promises";
import * as path from "path";
import * as os from "os";

const KEYRING_SERVICE = "ploydok";
const KEYRING_ACCOUNT = "system-signing-key";

/**
 * Load or generate system signing key.
 * Priority: OS keyring → fallback file → generate new.
 */
export async function loadOrGenerateSigningKey(): Promise<Buffer> {
  // Try OS keyring
  try {
    const keytar = await import("keytar");
    const stored = await keytar.getPassword(KEYRING_SERVICE, KEYRING_ACCOUNT);
    if (stored) {
      return Buffer.from(stored, "hex");
    }
  } catch (e) {
    // Keytar unavailable (headless env), continue to fallback
  }

  // Fallback file path
  const fallbackPath = getFallbackKeyPath();
  try {
    const content = await readFile(fallbackPath, "hex");
    return Buffer.from(content, "hex");
  } catch (e) {
    // File doesn't exist; generate new
  }

  // Generate new Ed25519 key
  const { generateKeyPairSync } = require("crypto");
  const { privateKey } = generateKeyPairSync("ed25519");
  const privKeyBuf = privateKey.export({ format: "raw" });

  // Store in keyring if available
  try {
    const keytar = await import("keytar");
    await keytar.setPassword(
      KEYRING_SERVICE,
      KEYRING_ACCOUNT,
      privKeyBuf.toString("hex")
    );
  } catch (e) {
    // Fallback to file storage
    const fallbackPath = getFallbackKeyPath();
    const dir = path.dirname(fallbackPath);
    await writeFile(fallbackPath, privKeyBuf.toString("hex"), { flag: "wx" });
    await chmod(fallbackPath, 0o400);
  }

  return privKeyBuf;
}

function getFallbackKeyPath(): string {
  if (process.env.NODE_ENV === "production") {
    return "/var/lib/ploydok/signing.key";
  } else {
    return path.join(os.homedir(), ".ploydok-dev", "signing.key");
  }
}
```

### `apps/api/src/signing/ed25519.ts`

```typescript
// SPDX-License-Identifier: AGPL-3.0-only

import { createHash, sign, verify } from "crypto";
import { AuditLog } from "@ploydok/db/schema";

const CANONICAL_VERSION = "v1";

/**
 * Canonicalize audit log entry as per spec.
 * Format: "v1\n<id>\n<created_at_iso>\n<user_id|->\n<action>\n<target_type>\n<target_id>\n<sha256_hex(metadata)>\n<prev_hash|->\n<hash>"
 */
export function canonicalPayload(
  entry: AuditLog,
  prevHash: string | null,
  nextHash: string
): string {
  const metadataHash = createHash("sha256")
    .update(JSON.stringify(entry.metadata || {}))
    .digest("hex");

  const lines = [
    CANONICAL_VERSION,
    entry.id,
    new Date(entry.createdAt).toISOString(),
    entry.userId || "-",
    entry.action,
    entry.targetType,
    entry.targetId,
    metadataHash,
    prevHash || "-",
    nextHash,
  ];

  return lines.join("\n");
}

/**
 * Sign a canonical payload with Ed25519 private key.
 */
export function signPayload(
  payload: string,
  privateKeyBuf: Buffer
): Buffer {
  const keyObject = require("crypto").createPrivateKey({
    key: privateKeyBuf,
    format: "raw",
    type: "ed25519",
  });

  return sign(null, Buffer.from(payload), keyObject);
}

/**
 * Verify a single signature.
 */
export function verifySignature(
  payload: string,
  signature: Buffer,
  publicKeyBuf: Buffer
): boolean {
  const keyObject = require("crypto").createPublicKey({
    key: publicKeyBuf,
    format: "raw",
    type: "ed25519",
  });

  try {
    return verify(null, Buffer.from(payload), keyObject, signature);
  } catch (e) {
    return false;
  }
}

/**
 * Verify a chain of audit log signatures.
 * Detects any tampering or broken links.
 */
export function verifyChain(
  entries: Array<{
    id: string;
    payload: string;
    signature: Buffer;
    publicKey: Buffer;
  }>
): { valid: boolean; firstBreakAt?: string; reason?: string } {
  for (const entry of entries) {
    if (!verifySignature(entry.payload, entry.signature, entry.publicKey)) {
      return {
        valid: false,
        firstBreakAt: entry.id,
        reason: "Signature verification failed",
      };
    }
  }

  // Verify hash chain continuity (prev_hash → current_hash)
  let prevHash: string | null = null;
  for (const entry of entries) {
    const lines = entry.payload.split("\n");
    const currentPrevHash = lines[8]; // prev_hash field
    const currentHash = lines[9]; // hash field

    if (prevHash && prevHash !== currentPrevHash) {
      return {
        valid: false,
        firstBreakAt: entry.id,
        reason: "Hash chain broken",
      };
    }

    prevHash = currentHash;
  }

  return { valid: true };
}

/**
 * Extract public key from Ed25519 private key.
 */
export function extractPublicKey(privateKeyBuf: Buffer): Buffer {
  const keyObject = require("crypto").createPrivateKey({
    key: privateKeyBuf,
    format: "raw",
    type: "ed25519",
  });

  const pubKeyObj = keyObject.export({ type: "spki", format: "der" });
  // Strip PKCS#8 wrapping to get raw 32-byte key
  // Ed25519 raw public key is last 32 bytes of SPKI DER
  return pubKeyObj.slice(-32);
}
```

## PR 3 — Worker + Anchor

### `apps/api/src/worker/handlers/audit-sign.ts`

```typescript
// SPDX-License-Identifier: AGPL-3.0-only

import { mkdir, appendFile } from "fs/promises";
import * as path from "path";
import * as os from "os";
import { createHash } from "crypto";
import { db } from "@ploydok/db";
import { auditLog, auditLogSignatures, signingKeys } from "@ploydok/db/schema";
import { eq, isNull, and } from "drizzle-orm";
import {
  canonicalPayload,
  signPayload,
  extractPublicKey,
} from "../signing/ed25519";
import { loadOrGenerateSigningKey } from "../signing/keyring";
import { ulid } from "ulid";
import { logger } from "../../logger";

export async function handleAuditSign() {
  logger.info("Starting audit-sign job");

  try {
    // Load system signing key
    const privKeyBuf = await loadOrGenerateSigningKey();
    const pubKey = extractPublicKey(privKeyBuf);

    // Ensure signing_keys entry exists
    const currentKey = await db
      .select()
      .from(signingKeys)
      .where(
        and(isNull(signingKeys.revokedAt), isNull(signingKeys.rotatedAt))
      )
      .limit(1);

    let keyId: string;
    if (currentKey.length === 0) {
      // Create new signing key record
      const year = new Date().getFullYear();
      const hashSuffix = createHash("sha256")
        .update(pubKey)
        .digest("hex")
        .slice(0, 4);
      keyId = `key_${year}_${hashSuffix}`;

      await db.insert(signingKeys).values({
        id: keyId,
        publicKey: pubKey,
        privateKeyKeyringRef: "keyring://ploydok/system-signing-key",
      });
    } else {
      keyId = currentKey[0].id;
    }

    // Fetch unsigned audit logs (ordered by id for hash continuity)
    const unsigned = await db
      .select()
      .from(auditLog)
      .where(isNull(auditLog.lastAnchoredAt))
      .orderBy(auditLog.id)
      .limit(1000); // Batch size

    if (unsigned.length === 0) {
      logger.debug("No unsigned audit logs");
      return;
    }

    let prevHash: string | null = null;
    const anchorLines: string[] = [];
    const signedEntries: typeof auditLogSignatures.$inferInsert[] = [];

    for (const entry of unsigned) {
      // Fetch previous hash from DB
      if (!prevHash) {
        const lastSigned = await db
          .select({ canonicalPayloadHash: auditLogSignatures.canonicalPayloadHash })
          .from(auditLogSignatures)
          .orderBy(auditLogSignatures.signedAt)
          .limit(1);
        prevHash = lastSigned[0]?.canonicalPayloadHash?.toString("hex") || null;
      }

      // Hash this entry
      const currentHash = createHash("sha256")
        .update(entry.id + JSON.stringify(entry))
        .digest("hex");

      // Build canonical payload
      const payload = canonicalPayload(entry, prevHash, currentHash);
      const signature = signPayload(payload, privKeyBuf);
      const payloadHash = createHash("sha256").update(payload).digest();

      // Record signature
      signedEntries.push({
        id: ulid(),
        auditId: entry.id,
        signatureV1Ed25519: signature,
        signerPublicKeyId: keyId,
        signerUserId: null,
        canonicalPayloadHash: payloadHash,
      });

      prevHash = currentHash;
    }

    // Batch insert signatures
    if (signedEntries.length > 0) {
      await db.insert(auditLogSignatures).values(signedEntries);
      logger.info(`Signed ${signedEntries.length} audit log entries`);

      // Update last_anchored_at on all signed entries
      const signedIds = unsigned.map((e) => e.id);
      await db
        .update(auditLog)
        .set({ lastAnchoredAt: new Date() })
        .where((t) => inArray(t.id, signedIds));
    }

    // Append to anchors.log (immutable append-only)
    const anchorsPath = getAnchorsPath();
    await mkdir(path.dirname(anchorsPath), { recursive: true });

    const timestamp = new Date().toISOString();
    const anchorLine = `${timestamp}|${prevHash || "-"}|${prevHash}|${unsigned.length}|${signedEntries.length}`;
    await appendFile(anchorsPath, anchorLine + "\n");

    logger.info(`Anchored ${signedEntries.length} signatures to ${anchorsPath}`);
  } catch (e) {
    logger.error("audit-sign job failed", { error: e });
    // Do not throw — allow the app to continue even if signing fails
    // Next cron will retry unsigned entries
  }
}

function getAnchorsPath(): string {
  if (process.env.NODE_ENV === "production") {
    return "/var/lib/ploydok/audit/anchors.log";
  } else {
    return path.join(os.homedir(), ".ploydok-dev", "audit", "anchors.log");
  }
}
```

## PR 4 — API route de vérification

### `apps/api/src/routes/audit.ts`

```typescript
// SPDX-License-Identifier: AGPL-3.0-only

import { Hono } from "hono";
import { z } from "zod";
import { db } from "@ploydok/db";
import { auditLog, auditLogSignatures, signingKeys } from "@ploydok/db/schema";
import { eq, gte, lte, isNotNull, and } from "drizzle-orm";
import { verifyChain } from "../signing/ed25519";
import { requireAdmin } from "../auth/require-scope";

const app = new Hono();

const verifyRequestSchema = z.object({
  from_id: z.string().optional(),
  to_id: z.string().optional(),
  validate_chain: z.boolean().optional().default(false),
});

app.post("/admin/audit/verify", requireAdmin, async (c) => {
  const body = verifyRequestSchema.parse(await c.req.json());

  try {
    // Fetch audit log entries in range
    const query = db
      .select()
      .from(auditLog)
      .innerJoin(
        auditLogSignatures,
        eq(auditLog.id, auditLogSignatures.auditId)
      )
      .innerJoin(signingKeys, eq(auditLogSignatures.signerPublicKeyId, signingKeys.id));

    let filtered = query;
    if (body.from_id) {
      filtered = filtered.where(gte(auditLog.id, body.from_id));
    }
    if (body.to_id) {
      filtered = filtered.where(lte(auditLog.id, body.to_id));
    }

    const results = await filtered.orderBy(auditLog.createdAt);

    // Rebuild entries for verification
    const entries = results.map((row) => ({
      id: row.audit_log.id,
      payload: canonicalPayload(
        row.audit_log,
        null, // Reconstruct from signatures table if needed
        "" // hash
      ),
      signature: row.audit_log_signatures.signatureV1Ed25519,
      publicKey: row.signing_keys.publicKey,
    }));

    let verification = { valid: true };
    if (body.validate_chain && entries.length > 0) {
      verification = verifyChain(entries);
    }

    // Aggregate public keys used
    const keyUsage = new Map<string, number>();
    results.forEach((row) => {
      const keyId = row.signing_keys.id;
      keyUsage.set(keyId, (keyUsage.get(keyId) || 0) + 1);
    });

    return c.json({
      valid: verification.valid,
      entries_checked: results.length,
      first_chain_break: verification.firstBreakAt
        ? { at: verification.firstBreakAt, reason: verification.reason }
        : null,
      public_key_summaries: Array.from(keyUsage.entries()).map(
        ([keyId, count]) => ({
          key_id: keyId,
          used_count: count,
        })
      ),
    });
  } catch (e) {
    return c.json({ error: e.message }, 500);
  }
});

export default app;
```
